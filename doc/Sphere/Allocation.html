def each_sec_on( secstep, &block )
  local = @from.localtime
  current = Time.local( 0, 0, 0, *local.to_a[ 3..-1 ] ) + ( ( local.hour * 3600.0 + local.min * 60 + local.sec + local.usec * 1e-6 ).div( secstep ) + 1 ) * secstep
  _each_sec( secstep, current, &block )
end
