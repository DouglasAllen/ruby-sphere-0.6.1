    Moon Phase in radian at a given Time

    # File lib/sphere/star.rb, line 520
def phase( time = nil )
  lb = lambdabeta( time )
  dl = lb.lambda - Sun.new( @time ).sl
  p = Math::acos( Math::cos( dl ) * Math::cos( lb.beta ) )
  t = ( dl / Math::PI ).floor
  sign = ( t % 2 == 0 ) ? +1 : -1
  offset = 2.0 * Math::PI * ( ( t.to_f / 2.0 ).ceil )
  offset + p * sign
end
