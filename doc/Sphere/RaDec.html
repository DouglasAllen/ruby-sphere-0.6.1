def to_lambdabeta( time = Time.new( 2000, 1, 1 ) )
  obl = Sphere::obliquity( time )
  LambdaBeta.new(                       atan2( sin( obl )*sin( @dec ) + cos( obl )*cos( @dec )*sin( @ra ), cos( @dec )*cos( @ra ) ),
                 asin( cos( obl )*sin( @dec ) - sin( obl )*cos( @dec )*sin( @ra ) )                      )
end
